167  size_t loc = get_location(key);
168  if (loc == this->myid) {
169  // data is local

172  if(it != _data.end()){
173  res.first = D_HASH_FOUND;
174  res.second=it->second;
175  return res;
176  }
177  else {
178  res.first=D_HASH_NOT_FOUND;
179  return res;
180  }
181  } else {
182  // FIXME:: rpc here; we need a convention of when the value returns
183  typename d_hash<Key,Value,Hash>::find_type return_value;
184  return_value.first = D_HASH_NOT_ARRIVED;
185  d_hash_args_get<Key,Value> da((intptr_t)&return_value, key, this->shared_id,this->myid);
186  this->rpcm.rpc(static_cast<int>(loc), _i_find_element<Key,Value,Hash>, da);
187  while (return_value.first == D_HASH_NOT_ARRIVED){
188  this->rpcm.advance();
189  }

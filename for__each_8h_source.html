259  edge_iterator eit = vit->find_edge(maxv);
260  if (eit != vit->edges_end()) {
261  edgeAdded = false;
262  } else {
263  vit->add_edge(maxv);
264  }
265  } else {

267  }
268  pthread_mutex_unlock(_locks+hashId);

319 template <class TG, class IG>
320 void make_task_graph(TG& tg, IG& ig) {
321  typedef typename IG::vertex_iterator vit_type;
322  typedef typename TG::vertex_iterator tg_vit_type;
323  typedef typename IG::edge_iterator eit_type;
324  typedef typename IG::vertexd_type vd_type;
325 
326  size_t NV = ig.num_vertices();
327 
328  // add vertices
329  for (vit_type vit = ig.vertices_begin(); vit != ig.vertices_end(); ++vit) {
330  tg.add_vertex(2*NV);
331  }
332  // and we add them one more time for reversed
333  for (vit_type vit = ig.vertices_begin(); vit != ig.vertices_end(); ++vit) {
334  if (vit->edges_size() != 0) {
335  // I take advantage of the property that the tg carry arround and
336  // not used
337  vd_type vd = tg.add_vertex(NV + vit->id());
338  tg_vit_type it = tg.find_vertex(vit->id());
339  // store the maping from reversed vertices to the initial ones
340  it->property()=vd;
341  } else {
342  // no succ;
343  tg_vit_type it = tg.find_vertex(vit->id());
344  it->property() = vit->id();
345  }
346  }
347  // now we add edges
348  for (vit_type vit = ig.vertices_begin(); vit != ig.vertices_end(); ++vit) {
349  for (eit_type eit = vit->edges_begin(); eit != vit->edges_end(); ++eit) {
350  tg.add_edge(vit->id(), eit->target());
351  }
352  }
353  // and finally reverse edges
354  size_t ep = 0;
355  for (vit_type vit = ig.vertices_begin(); vit != ig.vertices_end(); ++vit) {
356  for (eit_type eit = vit->preds_begin(); eit != vit->preds_end(); ++eit) {
357  tg_vit_type it_s = tg.find_vertex(vit->id());
358  tg_vit_type it_t = tg.find_vertex(eit->source());
359  vd_type src;
360 
361  if (it_s->property() == -1)
362  src = it_s->id();
363  else
364  src = it_s->property();
365 
366  tg.add_edge(src, it_t->property(), ep);
367  }
368  }
